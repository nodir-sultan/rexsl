48  /**
49   * Validates CSS files against style sheet rules.
50   *
51   * <p>The class is immutable and thread-safe.
52   *
53   * <p>Since this class is NOT public its documentation is not available online.
54   * All details of this check should be explained in the JavaDoc of
55   * {@link DefaultChecksProvider}.
56   *
57   * <p>{@code CssLint.class} file is created by Rhino JavaScript-to-Java
58   * compiler during {@code process-sources} Maven phase (see {@code pom.xml}
59   * file). Here we're just executing this class in a standalone process, in
60   * order to capture its output into string.
61   *
62   * @author Dmitry Bashkin (dmitry.bashkin@rexsl.com)
63   * @author Yegor Bugayenko (yegor@tpc2.com)
64   * @version $Id$
65   * @see <a href="http://www.mozilla.org/rhino/jsc.html">Rhino JavaScript to Java compiler</a>
66   * @see <a href="https://github.com/stubbornella/csslint/wiki/Command-line-interface">CSSLint Command Line Interface</a>
67   */
68  @ToString
69  @EqualsAndHashCode
70  final class CssStaticCheck implements Check {
71  
72      @Override
73      @Loggable(Loggable.DEBUG)
74      public void setScope(@NotNull final String scope) {
75          // nothing to scope here
76      }
77  
78      @Override
79      @Loggable(Loggable.DEBUG)
80      public boolean validate(@NotNull final Environment env) throws IOException {
81          final File csslint = new File(
82              this.getClass().getResource("/CssLint.class").getFile()
83          );
84          final ProcessBuilder builder = new ProcessBuilder(
85              "java",
86              "-classpath",
87              StringUtils.join(
88                  Sets.union(
89                      env.classpath(false),
90                      Sets.newHashSet(
91                          // @checkstyle MultipleStringLiterals (1 line)
92                          new File("."),
93                          this.jar(org.mozilla.javascript.Script.class)
94                      )
95                  ),
96                  System.getProperty("path.separator")
97              ),
98              "CssLint",
99              env.basedir().getPath(),
100             "--format=compact"
101         );
102         builder.directory(csslint.getParentFile());
103         final String report;
104         try {
105             final Process process = builder.start();
106             if (process.waitFor() != 0) {
107                 throw new IllegalStateException(
108                     IOUtils.toString(process.getErrorStream())
109                 );
110             }
111             report = IOUtils.toString(process.getInputStream());
112         } catch (InterruptedException ex) {
113             Thread.currentThread().interrupt();
114             throw new IllegalStateException(ex);
115         }
116         return this.isValid(report);
117     }
118 
119     /**
120      * Get address of our JAR or directory.
121      * @param resource Name of resource
122      * @return The file
123      */
124     private File jar(final Class<?> resource) {
125         final String name = resource.getName()
126             // @checkstyle MultipleStringLiterals (1 line)
127             .replace(".", System.getProperty("file.separator"));
128         final URL res = this.getClass().getResource(
129             String.format("/%s.class", name)
130         );
131         if (res == null) {
132             throw new IllegalStateException(
133                 String.format(
134                     "can't find JAR for %s",
135                     name
136                 )
137             );
138         }
139         final String path = res.getFile().replaceAll("\\!.*$", "");
140         File file;
141         if ("jar".equals(FilenameUtils.getExtension(path))) {
142             file = new File(URI.create(path).getPath());
143         } else {
144             file = new File(path).getParentFile()
145                 .getParentFile()
146                 .getParentFile()
147                 .getParentFile();
148         }
149         Logger.debug(this, "#jar(%s): found at %s", resource.getName(), file);
150         return file;
151     }

153     /**
154      * Validate the report from CSSLint.
155      *
156      * <p>CSSLint report is a plain text document, where every line is a
157      * message about one defect. In this method we split this plain text
158      * document into lines and fetch them one by one to the log. If there are
159      * no lines in the document we return {@code true}, which means that
160      * there are no errors and the CSS document is valid.
161      *
162      * @param report The report (one problem per line)
163      * @return True if it's valid (no errors)
164      */
165     private boolean isValid(final String report) {
166         final boolean valid;
167         if (report.contains("Lint Free!")) {
168             valid = true;
169         } else {
170             final String[] lines = report.split("\n");
171             for (final String line : lines) {
172                 Logger.warn(this, "%s", line.trim());
173             }
174             valid = lines.length == 0;
175         }
176         return valid;
177     }
