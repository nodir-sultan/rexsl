45  /**
46   * Provider of checks.
47   *
48   * <p>This class is NOT thread-safe.
49   *
50   * <p>Checks are executed in the order they are listed here.
51   *
52   * <h3>BinaryFilesCheck</h3>
53   *
54   * <p>You shouldn't keep any binary files (images, video, etc.) in
55   * {@code src/main/webapp} folder and deploy them inside your {@code WAR}
56   * package. Instead, you should use some storage service
57   * (<a href="http://aws.amazon.com/s3">Amazon S3</a>, for example) and a
58   * content delivery network (CDN) on top of it
59   * (<a href="http://aws.amazon.com/cloudfront">Amazon CloudFront</a>,
60   * for example). This check validates that no binary files are located inside
61   * {@code src/main/webapp} and complains if anything found.
62   *
63   * <h3>JigsawCssCheck</h3>
64   *
65   * <p>All CSS stylesheets from {@code src/main/webapp/css} folder are sent
66   * to W3C Jigsaw validator. Any defect found will fail the build. Read more
67   * about this validation in {@link com.rexsl.w3c.Validator}.
68   *
69   * <h3>JSStaticCheck</h3>
70   *
71   * <p>JavaScript files from {@code src/main/webapp/js} are validated with YUI.
72   * This test is not implemented yet.
73   *
74   * <h3>FilesStructureCheck</h3>
75   *
76   * <p>The following structure of directories and files is enforced inside
77   * {@code src/main/webapp} and {@code src/test/rexsl}
78   * (all of them should be present):
79   *
80   * <pre>
81   * src/main/webapp
82   *   /xsl
83   *   /WEB-INF/web.xml
84   *   /robots.txt
85   * src/test/rexsl
86   *   /xml
87   *   /scripts
88   *   /xsd
89   *   /xhtml
90   * </pre>
91   *
92   * <h3>RexslFilesCheck</h3>
93   *
94   * <p>Test folders should contain files of certain types only:
95   *
96   * <pre>
97   * src/test/rexsl
98   *   /xml/*.xml
99   *   /scripts/*.groovy
100  *   /xsd/*.xsd
101  *   /xhtml/*.groovy
102  *   /bootstrap/*.groovy
103  *   /setup/*.groovy
104  * </pre>
105  *
106  * <p>If any other types of files are found in these folders the check will
107  * complain and fail the build.
108  *
109  * <h3>LibrariesCheck</h3>
110  *
111  * <p>The check verifies all libraries from {@code target/.../WEB-INF/lib}
112  * and finds conflicts between them.
113  *
114  * <h3>XhtmlOutputCheck</h3>
115  *
116  * <p>XML pages from {@code src/test/rexsl/xml} are transformed to XHTML using
117  * XSL stylesheets attatched to them. Then, produced HTML pages are sent to
118  * W3C validator. And then, they are sent to designated groovy scripts
119  * from {@code src/test/rexsl/xhtml}. Name of Groovy script should be identical
120  * to the name of the XML page (except the extension, of course). Keep in mind
121  * that Groovy pre-compiles scripts to Java classes, that's why names should
122  * obey Java class naming rules.
123  *
124  * <p>Content of a created XHTML page is sent to the Groovy script in
125  * {@code rexsl.document} variable. It's very convenient to use
126  * {@link com.rexsl.test.XhtmlMatchers} methods to assert certain XPath
127  * queries on the HTML documents.
128  *
129  * <h3>InContainerScriptsCheck</h3>
130  *
131  * <p>The check starts an embedded Java Servlet container and runs Groovy
132  * scripts from {@code src/test/rexsl/scripts}, one by one. Scripts can
133  * use {@link com.jcabi.http.request.ApacheRequest} class (or similar)
134  * to test the application
135  * through its RESTful interface. The URI of the running application is
136  * supplied in {@code rexsl.home} variable.
137  *
138  * <h3>WebXmlCheck</h3>
139  *
140  * <p>The check validates {@code src/main/webapp/WEB-INF/web.xml} XML file
141  * against its XSD schema and fails the build if any incosistencies are
142  * detected. Validation may take some time (15-45 seconds).
143  *
144  * <p>The class is thread-safe.
145  *
146  * @author Yegor Bugayenko (yegor@tpc2.com)
147  * @version $Id$
148  * @checkstyle ClassDataAbstractionCoupling (100 lines)
149  */
150 @ToString
151 @EqualsAndHashCode(of = "test")
152 public final class DefaultChecksProvider implements ChecksProvider {
153 
154     /**
155      * All known checks.
156      */
157     private static final Collection<String> NAMES =
158         Collections.unmodifiableList(
159             Arrays.asList(
160                 "com.rexsl.maven.checks.BinaryFilesCheck",
161                 "com.rexsl.maven.checks.JigsawCssCheck",
162                 "com.rexsl.maven.checks.JSStaticCheck",
163                 "com.rexsl.maven.checks.FilesStructureCheck",
164                 "com.rexsl.maven.checks.RexslFilesCheck",
165                 "com.rexsl.maven.checks.LibrariesCheck",
166                 "com.rexsl.maven.checks.XhtmlOutputCheck",
167                 "com.rexsl.maven.checks.InContainerScriptsCheck",
168                 "com.rexsl.maven.checks.WebXmlCheck"
169             )
170         );
171 
172     /**
173      * Checks to retrieve (comma-separated list of full or short class names).
174      */
175     private final transient Set<String> checks = new LinkedHashSet<String>(
176         DefaultChecksProvider.NAMES
177     );
178 
179     /**
180      * Test scope.
181      */
182     private transient String test = "";
183 
184     /**
185      * Mutex for synchronization.
186      */
187     @SuppressWarnings("PMD.FinalFieldCouldBeStatic")
188     private final transient Object mutex = "_mutex_";
189 
190     @Override
191     @Loggable(Loggable.DEBUG)
192     public Collection<Check> all() {
193         final Set<Check> all = new LinkedHashSet<Check>(0);
194         for (final String name : this.checks) {
195             try {
196                 all.add(this.build(name));
197             } catch (InvalidCheckException ex) {
198                 Logger.info(this, "Can't find %s, available checks:", name);
199                 for (final String check : DefaultChecksProvider.NAMES) {
200                     Logger.info(this, "  %s", check);
201                 }
202                 throw new IllegalArgumentException(
203                     String.format("Check %s can't be built", name),
204                     ex
205                 );
206             }
207         }
208         return Collections.unmodifiableSet(all);
209     }
210 
211     @Override
212     @Loggable(Loggable.DEBUG)
213     public void setTest(@NotNull final String scope) {
214         synchronized (this.mutex) {
215             this.test = scope;
216         }
217     }
218 
219     @Override
220     @Loggable(Loggable.DEBUG)
221     public void setCheck(@NotNull final String scope) {
222         synchronized (this.mutex) {
223             this.checks.clear();
224             this.checks.addAll(
225                 Arrays.asList(scope.split("\\s*,\\s*"))
226             );
227         }
228     }
229 
230     /**
231      * Build check from its name.
232      * @param name Full name of the class or a short one
233      * @return The check
234      * @throws DefaultChecksProvider.InvalidCheckException If failed
235      */
236     private Check build(final String name)
237         throws DefaultChecksProvider.InvalidCheckException {
238         final String cname;
239         if (name.contains(".")) {
240             cname = name;
241         } else {
242             cname = String.format(
243                 "%s.%s",
244                 DefaultChecksProvider.class.getPackage().getName(),
245                 name
246             );
247         }
248         final Check check;
249         try {
250             check = Check.class.cast(Class.forName(cname).newInstance());
251         } catch (ClassNotFoundException ex) {
252             throw new DefaultChecksProvider.InvalidCheckException(ex);
253         } catch (InstantiationException ex) {
254             throw new DefaultChecksProvider.InvalidCheckException(ex);
255         } catch (IllegalAccessException ex) {
256             throw new DefaultChecksProvider.InvalidCheckException(ex);
257         }
258         check.setScope(this.test);
259         return check;
260     }
261 
262     /**
263      * Thrown when class can't be built.
264      */
265     private static final class InvalidCheckException extends Exception {
266         /**
267          * Serialization marker.
268          */
269         private static final long serialVersionUID = 0x7526FA7CDAA2147AL;
270         /**
271          * Public ctor.
272          * @param cause Cause of the problem
273          */
274         public InvalidCheckException(final Throwable cause) {
275             super(cause);
276         }
277     }
