50  /**
51   * Package resources.
52   *
53   * <p>Text resources should be packed before they got delivered to the
54   * {@code WAR} package. For example, comments should be removed from CSS and
55   * XSL stylesheets, JavaScripts should be minified, etc. We have a number of
56   * convenient packers that do this job, you just need to call them. First of
57   * a {@code package} goal should be called:
58   *
59   * <pre>&lt;plugin>
60   *   &lt;groupId&gt;com.rexsl&lt;/groupId&gt;
61   *   &lt;artifactId&gt;rexsl-maven-plugin&lt;/artifactId&gt;
62   *   &lt;version&gt;...&lt;/version&gt;
63   *   &lt;executions&gt;
64   *     &lt;execution&gt;
65   *       &lt;phase&gt;&lt;phase&gt;
66   *       &lt;goals&gt;
67   *         &lt;goal&gt;package&lt;/goal&gt;
68   *       &lt;/goals&gt;
69   *     &lt;/execution&gt;
70   *   &lt;/executions&gt;
71   * &lt;/plugin&gt;</pre>
72   *
73   * <p>The goal will package all files it knows about and place their packaged
74   * (minified) versions into {@code webappDirectory}. Then, you should configure
75   * {@code maven-war-plugin} to not touch these files again (to avoid their
76   * packaging):
77   *
78   * <pre>&lt;plugin>
79   *   &lt;artifactId&gt;maven-war-plugin&lt;/artifactId&gt;
80   *   &lt;configuration&gt;
81   *     &lt;warSourceExcludes&gt;js/**, css/**, xsl/**&lt;/warSourceExcludes&gt;
82   *     [..]
83   *   &lt;/configuration&gt;
84   * &lt;/plugin&gt;</pre>
85   *
86   * @author Yegor Bugayenko (yegor@tpc2.com)
87   * @version $Id$
88   */
89  @Mojo(
90      name = "package", defaultPhase = LifecyclePhase.PROCESS_RESOURCES,
91      threadSafe = true
92  )
93  public final class PackageMojo extends AbstractRexslMojo {
94  
95      /**
96       * Static resources (CSS, XSL, JS, etc.) should be filtered before
97       * packaging.
98       *
99       * <p>At the moment only properties defined in {@code pom/properties}
100      * section are supported.
101      *
102      * @since 0.3.7
103      * @see <a href="http://trac.rexsl.com/rexsl/ticket/342">Introduced in ticket #342</a>
104      */
105     @Parameter(property = "rexsl.filtering")
106     private transient boolean filtering;
107 
108     /**
109      * Set filtering.
110      * @param fltr Shall we filter before packaging?
111      */
112     public void setFiltering(final boolean fltr) {
113         this.filtering = fltr;
114     }
115 
116     @Override
117     protected void run() {
118         Filter filter;
119         if (this.filtering) {
120             filter = new Filter() {
121                 @Override
122                 public Reader filter(final File file) throws IOException {
123                     return new InputStreamReader(
124                         new FileInputStream(file),
125                         CharEncoding.UTF_8
126                     );
127                 }
128             };
129         } else {
130             filter = new PackageMojo.PropsFilter();
131         }
132         final long start = System.currentTimeMillis();
133         final Set<Packer> packers = new PackersProvider().all();
134         for (final Packer packer : packers) {
135             packer.pack(this.env(), filter);
136         }
137         Logger.info(
138             this,
139             "Packaging finished in %[ms]s",
140             System.currentTimeMillis() - start
141         );
142     }
143 
144     /**
145      * Filtering using project properties.
146      * @todo #342 Only properties from PROPERTIES section in pom.xml are
147      *  supported. I don't know how to get access to all other project props.
148      */
149     private final class PropsFilter implements Filter {
150         @Override
151         public Reader filter(final File file) throws IOException {
152             final String origin = FileUtils.readFileToString(file);
153             final StringBuffer text = new StringBuffer();
154             final Matcher matcher = Pattern.compile("\\$\\{([\\w\\.\\-]+)\\}")
155                 .matcher(origin);
156             final Properties props = PackageMojo.this.project().getProperties();
157             Logger.debug(
158                 this,
159                 "#filter(..): all properties available %[list]s",
160                 props.keySet()
161             );
162             while (matcher.find()) {
163                 final String name = matcher.group(1);
164                 String replacer;
165                 if (props.containsKey(name)) {
166                     replacer = props.get(name).toString();
167                     Logger.info(
168                         this,
169                         "'%s' replaced with '%s' in %s",
170                         matcher.group(),
171                         replacer,
172                         file
173                     );
174                 } else {
175                     replacer = matcher.group();
176                     Logger.warn(
177                         this,
178                         "'%s' can't be replaced in %s (not found in pom.xml)",
179                         matcher.group(),
180                         file
181                     );
182                 }
183                 matcher.appendReplacement(
184                     text,
185                     Matcher.quoteReplacement(replacer)
186                 );
187             }
188             matcher.appendTail(text);
189             return new StringReader(text.toString());
190         }
191     }
