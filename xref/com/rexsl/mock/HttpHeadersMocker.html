49  /**
50   * Builds an instance of {@link HttpHeaders}.
51   * @author Yegor Bugayenko (yegor@tpc2.com)
52   * @version $Id$
53   */
54  public final class HttpHeadersMocker {
55  
56      /**
57       * The mock.
58       */
59      private final transient HttpHeaders subj =
60          Mockito.mock(HttpHeaders.class);
61  
62      /**
63       * Multivalued map of headers.
64       */
65      private final transient ConcurrentMap<String, List<String>> headers =
66          new ConcurrentSkipListMap<String, List<String>>();
67  
68      /**
69       * Public ctor.
70       */
71      @SuppressWarnings("unchecked")
72      public HttpHeadersMocker() {
73          final MultivaluedMap<String, String> map =
74              Mockito.mock(MultivaluedMap.class);
75          Mockito.doReturn(map).when(this.subj).getRequestHeaders();
76          Mockito.doAnswer(
77              new Answer<List<String>>() {
78                  @Override
79                  public List<String> answer(final InvocationOnMock inv) {
80                      final String name = inv.getArguments()[0].toString();
81                      return HttpHeadersMocker.this.headers.get(name);
82                  }
83              }
84          ).when(this.subj).getRequestHeader(Mockito.anyString());
85          Mockito.doAnswer(
86              new Answer<String>() {
87                  @Override
88                  public String answer(final InvocationOnMock inv) {
89                      final String name = inv.getArguments()[0].toString();
90                      String first = null;
91                      if (HttpHeadersMocker.this.headers.containsKey(name)) {
92                          final Iterator<String> values =
93                              HttpHeadersMocker.this.headers.get(name).iterator();
94                          if (values.hasNext()) {
95                              first = values.next();
96                          }
97                      }
98                      return first;
99                  }
100             }
101         ).when(map).getFirst(Mockito.anyString());
102         Mockito.doAnswer(
103             new Answer<Boolean>() {
104                 @Override
105                 public Boolean answer(final InvocationOnMock inv) {
106                     return HttpHeadersMocker.this.headers.containsKey(
107                         inv.getArguments()[0].toString()
108                     );
109                 }
110             }
111         ).when(map).containsKey(Mockito.anyString());
112         Mockito.doAnswer(
113             new Answer<Set<Map.Entry<String, List<String>>>>() {
114                 @Override
115                 public Set<Map.Entry<String, List<String>>> answer(
116                     final InvocationOnMock inv) {
117                     return HttpHeadersMocker.this.headers.entrySet();
118                 }
119             }
120         ).when(map).entrySet();
121         Mockito.doAnswer(
122             new Answer<Set<String>>() {
123                 @Override
124                 public Set<String> answer(final InvocationOnMock inv) {
125                     return HttpHeadersMocker.this.headers.keySet();
126                 }
127             }
128         ).when(map).keySet();
129         Mockito.doAnswer(
130             new Answer<List<String>>() {
131                 @Override
132                 public List<String> answer(final InvocationOnMock inv) {
133                     return HttpHeadersMocker.this.headers.get(
134                         inv.getArguments()[0].toString()
135                     );
136                 }
137             }
138         ).when(map).get(Mockito.anyString());
139     }
140 
141     /**
142      * With this header on board.
143      * @param name The name of it
144      * @param value The value of it
145      * @return This object
146      */
147     public HttpHeadersMocker withHeader(final String name, final String value) {
148         this.headers.putIfAbsent(name, new LinkedList<String>());
149         this.headers.get(name).add(value);
150         return this;
151     }
152 
153     /**
154      * Build an instance of provided class.
155      * @return The resource just created
156      */
157     public HttpHeaders mock() {
158         Mockito.doAnswer(
159             // @checkstyle AnonInnerLength (50 lines)
160             new Answer<Map<String, Cookie>>() {
161                 @Override
162                 public Map<String, Cookie> answer(final InvocationOnMock inv) {
163                     final ConcurrentMap<String, Cookie> cookies =
164                         new ConcurrentHashMap<String, Cookie>();
165                     final Collection<String> hdrs =
166                         HttpHeadersMocker.this.headers.get(HttpHeaders.COOKIE);
167                     if (hdrs != null) {
168                         for (final String header : hdrs) {
169                             for (final HttpCookie cookie
170                                 : HttpCookie.parse(header)) {
171                                 cookies.put(
172                                     cookie.getName(),
173                                     new Cookie(
174                                         cookie.getName(),
175                                         cookie.getValue()
176                                     )
177                                 );
178                             }
179                         }
180                     }
181                     return cookies;
182                 }
183             }
184         ).when(this.subj).getCookies();
185         return this.subj;
186     }
