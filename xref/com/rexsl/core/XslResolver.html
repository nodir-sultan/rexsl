61  /**
62   * Provider of JAXB {@link Marshaller} for JAX-RS framework.
63   *
64   * <p>You don't need to use this class directly. It is made public only becuase
65   * JAX-RS implementation should be able to discover it in classpath.
66   *
67   * @author Yegor Bugayenko (yegor@tpc2.com)
68   * @author Krzysztof Krason (Krzysztof.Krason@gmail.com)
69   * @version $Id$
70   * @since 0.2
71   */
72  @ToString
73  @EqualsAndHashCode(of = { "folder", "classes", "context" })
74  @Provider
75  @Produces({ MediaType.APPLICATION_XML, MediaType.TEXT_XML })
76  @Loggable(Loggable.DEBUG)
77  public final class XslResolver implements ContextResolver<Marshaller> {
78  
79      /**
80       * Folder name of the location of XSD files
81       * (name of {@link ServletContext} init parameter).
82       */
83      public static final String XSD_FOLDER = "com.rexsl.core.XSD_FOLDER";
84  
85      /**
86       * Manifest xsl suffix entry name.
87       */
88      private static final String XSL_SUFFIX = "ReXSL-Suffix";
89  
90      /**
91       * Folder with XSD files.
92       * @see #setServletContext(ServletContext)
93       */
94      private transient File folder;
95  
96      /**
97       * Classes to process.
98       */
99      private final transient Set<Class<?>> classes = new HashSet<Class<?>>(0);
100 
101     /**
102      * JAXB context.
103      */
104     private transient JAXBContext context;
105 
106     /**
107      * Servlet request.
108      */
109     private transient HttpServletRequest request;
110 
111     /**
112      * Set servlet context from container, to be called by JAX-RS framework
113      * because of {@link Context} annotation.
114      * @param ctx The context
115      */
116     @Context
117     public void setServletContext(@NotNull final ServletContext ctx) {
118         final String name = ctx.getInitParameter(XslResolver.XSD_FOLDER);
119         if (name != null) {
120             this.folder = new File(name);
121             Logger.debug(
122                 this,
123                 "#setServletContext(%s): XSD folder set to '%s'",
124                 ctx.getClass().getName(),
125                 this.folder
126             );
127         }
128     }
129 
130     /**
131      * Set request to provide information about resourse context.
132      * @param req The request
133      */
134     @Context
135     public void setHttpServletRequest(@NotNull final HttpServletRequest req) {
136         this.request = req;
137     }
138 
139     /**
140      * {@inheritDoc}
141      *
142      * <p>JAXBContext is thread-safe, that's why we don't synchronize here.
143      *
144      * @see <a href="http://jaxb.java.net/guide/Performance_and_thread_safety.html">JAXBContext is thread-safe</a>
145      */
146     @Override
147     @NotNull
148     public Marshaller getContext(@NotNull final Class<?> type) {
149         Marshaller mrsh;
150         try {
151             mrsh = this.buildContext(type).createMarshaller();
152             mrsh.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
153             final String header = Logger.format(
154                 "\n<?xml-stylesheet type='text/xsl' href='%s'?>",
155                 StringEscapeUtils.escapeXml11(this.stylesheet(type))
156             );
157             mrsh.setProperty("com.sun.xml.bind.xmlHeaders", header);
158         } catch (final JAXBException ex) {
159             throw new IllegalStateException(ex);
160         }
161         if (this.folder == null) {
162             Logger.debug(
163                 this,
164                 "#getContext(%s): marshaller created (no XSD validator)",
165                 type.getName()
166             );
167         } else {
168             this.addXsdValidatorToMarshaller(mrsh, type);
169         }
170         return mrsh;
171     }
172 
173     /**
174      * Add new class to context.
175      * @param cls The class we should add
176      */
177     public void add(@NotNull final Class<?> cls) {
178         synchronized (this.classes) {
179             if (!this.classes.contains(cls)) {
180                 try {
181                     this.classes.add(cls);
182                     this.context = JAXBContext.newInstance(
183                         this.classes.toArray(new Class<?>[this.classes.size()])
184                     );
185                     Logger.info(
186                         this,
187                         // @checkstyle LineLength (1 line)
188                         "#add(%s): added to JAXBContext (%d total), stylesheet: '%s'",
189                         cls.getName(),
190                         this.classes.size(),
191                         this.stylesheet(cls)
192                     );
193                 } catch (final JAXBException ex) {
194                     throw new IllegalStateException(ex);
195                 }
196             }
197         }
198     }
199 
200     /**
201      * Create and return a context.
202      * @param cls The class we should process
203      * @return The context
204      */
205     private JAXBContext buildContext(final Class<?> cls) {
206         this.add(cls);
207         return this.context;
208     }
209 
210     /**
211      * Returns the name of XSL stylesheet for this type.
212      * @param type The class
213      * @return The name of stylesheet
214      * @see #getContext(Class)
215      */
216     private String stylesheet(final Class<?> type) {
217         final Annotation antn = type.getAnnotation(Stylesheet.class);
218         String stylesheet;
219         if (antn == null) {
220             stylesheet = Logger.format(
221                 "/xsl/%s.xsl",
222                 type.getSimpleName()
223             );
224             if (this.request != null) {
225                 try {
226                     stylesheet = new URL(
227                         this.request.getScheme(),
228                         this.request.getServerName(),
229                         this.request.getServerPort(),
230                         Logger.format(
231                             "%s%s%s",
232                             this.request.getContextPath(),
233                             stylesheet,
234                             this.suffix()
235                         )
236                     ).toString();
237                 } catch (final MalformedURLException ex) {
238                     throw new IllegalStateException(ex);
239                 }
240             }
241         } else {
242             stylesheet = ((Stylesheet) antn).value();
243         }
244         return stylesheet;
245     }
246 
247     /**
248      * Suffix for XSL files.
249      * @return Query string with suffix or empty string.
250      */
251     private String suffix() {
252         final String suffix;
253         if (Manifests.exists(XslResolver.XSL_SUFFIX)) {
254             suffix = String.format(
255                 "?%s",
256                 Manifests.read(XslResolver.XSL_SUFFIX)
257             );
258         } else {
259             suffix = "";
260         }
261         return suffix;
262     }
263 
264     /**
265      * Configure marhaller adding a xsd validator.
266      * @param mrsh The marshaller, already created and ready to marshal
267      * @param type The class to be marshalled
268      * @see #getContext(Class)
269      */
270     private void addXsdValidatorToMarshaller(final Marshaller mrsh,
271         final Class<?> type) {
272         final String name = XslResolver.schema(type);
273         if (name.isEmpty()) {
274             Logger.debug(
275                 this,
276                 "Schema validation turned off for class '%s'",
277                 type.getName()
278             );
279         } else {
280             final File xsd = new File(this.folder, name);
281             if (xsd.exists()) {
282                 final SchemaFactory factory = SchemaFactory.newInstance(
283                     XMLConstants.W3C_XML_SCHEMA_NS_URI
284                 );
285                 try {
286                     mrsh.setSchema(factory.newSchema(xsd));
287                 } catch (final SAXException ex) {
288                     throw new IllegalStateException(
289                         Logger.format(
290                             "Failed to use XSD schema from '%s' for class '%s'",
291                             xsd,
292                             type.getName()
293                         ),
294                         ex
295                     );
296                 }
297                 try {
298                     mrsh.setEventHandler(new XsdEventHandler());
299                 } catch (final JAXBException ex) {
300                     throw new IllegalStateException(ex);
301                 }
302                 Logger.debug(
303                     this,
304                     "Class '%s' will be validated with '%s' schema",
305                     type.getName(),
306                     xsd
307                 );
308             } else {
309                 Logger.warn(
310                     this,
311                     "No XSD schema for class '%s' in '%s' file",
312                     type.getName(),
313                     xsd
314                 );
315             }
316         }
317     }
318 
319     /**
320      * Returns the name of XSD schema for this type.
321      * @param type The class
322      * @return The name of XSD file
323      */
324     private static String schema(final Class<?> type) {
325         final Annotation antn = type.getAnnotation(Schema.class);
326         final String schema;
327         if (antn == null) {
328             schema = Logger.format("%s.xsd", type.getName());
329         } else {
330             if (((Schema) antn).ignore()) {
331                 schema = "";
332             } else {
333                 schema = ((Schema) antn).value();
334             }
335         }
336         return schema;
337     }
