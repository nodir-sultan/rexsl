60  /**
61   * Provider of JAXB {@link Marshaller} for JAX-RS framework.
62   *
63   * <p>You don't need to use this class directly. It is made public only becuase
64   * JAX-RS implementation should be able to discover it in classpath.
65   *
66   * @author Yegor Bugayenko (yegor@tpc2.com)
67   * @author Krzysztof Krason (Krzysztof.Krason@gmail.com)
68   * @version $Id$
69   * @since 0.2
70   */
71  @ToString
72  @EqualsAndHashCode(of = { "folder", "classes", "context" })
73  @Provider
74  @Produces({ MediaType.APPLICATION_XML, MediaType.TEXT_XML })
75  @Loggable(Loggable.DEBUG)
76  public final class XslResolver implements ContextResolver<Marshaller> {
77  
78      /**
79       * Folder name of the location of XSD files
80       * (name of {@link ServletContext} init parameter).
81       */
82      public static final String XSD_FOLDER = "com.rexsl.core.XSD_FOLDER";
83  
84      /**
85       * Folder with XSD files.
86       * @see #setServletContext(ServletContext)
87       */
88      private transient File folder;
89  
90      /**
91       * Classes to process.
92       */
93      private final transient Set<Class<?>> classes = new HashSet<Class<?>>(0);
94  
95      /**
96       * JAXB context.
97       */
98      private transient JAXBContext context;
99  
100     /**
101      * Servlet request.
102      */
103     private transient HttpServletRequest request;
104 
105     /**
106      * Set servlet context from container, to be called by JAX-RS framework
107      * because of {@link Context} annotation.
108      * @param ctx The context
109      */
110     @Context
111     public void setServletContext(@NotNull final ServletContext ctx) {
112         final String name = ctx.getInitParameter(XslResolver.XSD_FOLDER);
113         if (name != null) {
114             this.folder = new File(name);
115             Logger.debug(
116                 this,
117                 "#setServletContext(%s): XSD folder set to '%s'",
118                 ctx.getClass().getName(),
119                 this.folder
120             );
121         }
122     }
123 
124     /**
125      * Set request to provide information about resourse context.
126      * @param req The request
127      */
128     @Context
129     public void setHttpServletRequest(@NotNull final HttpServletRequest req) {
130         this.request = req;
131     }

133     /**
134      * {@inheritDoc}
135      *
136      * <p>JAXBContext is thread-safe, that's why we don't synchronize here.
137      *
138      * @see <a href="http://jaxb.java.net/guide/Performance_and_thread_safety.html">JAXBContext is thread-safe</a>
139      */
140     @Override
141     @NotNull
142     public Marshaller getContext(@NotNull final Class<?> type) {
143         Marshaller mrsh;
144         try {
145             mrsh = this.buildContext(type).createMarshaller();
146             mrsh.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
147             final String header = Logger.format(
148                 "\n<?xml-stylesheet type='text/xsl' href='%s'?>",
149                 StringEscapeUtils.escapeXml(this.stylesheet(type))
150             );
151             mrsh.setProperty("com.sun.xml.bind.xmlHeaders", header);
152         } catch (JAXBException ex) {
153             throw new IllegalStateException(ex);
154         }
155         if (this.folder == null) {
156             Logger.debug(
157                 this,
158                 "#getContext(%s): marshaller created (no XSD validator)",
159                 type.getName()
160             );
161         } else {
162             this.addXsdValidatorToMarshaller(mrsh, type);
163         }
164         return mrsh;
165     }
166 
167     /**
168      * Add new class to context.
169      * @param cls The class we should add
170      */
171     public void add(@NotNull final Class<?> cls) {
172         synchronized (this.classes) {
173             if (!this.classes.contains(cls)) {
174                 try {
175                     this.classes.add(cls);
176                     this.context = JAXBContext.newInstance(
177                         this.classes.toArray(new Class<?>[this.classes.size()])
178                     );
179                     Logger.info(
180                         this,
181                         // @checkstyle LineLength (1 line)
182                         "#add(%s): added to JAXBContext (%d total), stylesheet: '%s'",
183                         cls.getName(),
184                         this.classes.size(),
185                         this.stylesheet(cls)
186                     );
187                 } catch (JAXBException ex) {
188                     throw new IllegalStateException(ex);
189                 }
190             }
191         }
192     }
193 
194     /**
195      * Create and return a context.
196      * @param cls The class we should process
197      * @return The context
198      */
199     private JAXBContext buildContext(final Class<?> cls) {
200         this.add(cls);
201         return this.context;
202     }

204     /**
205      * Returns the name of XSL stylesheet for this type.
206      * @param type The class
207      * @return The name of stylesheet
208      * @see #getContext(Class)
209      */
210     private String stylesheet(final Class<?> type) {
211         final Annotation antn = type.getAnnotation(Stylesheet.class);
212         String stylesheet;
213         if (antn == null) {
214             stylesheet = Logger.format(
215                 "/xsl/%s.xsl",
216                 type.getSimpleName()
217             );
218             if (this.request != null) {
219                 try {
220                     stylesheet = new URL(
221                         this.request.getScheme(),
222                         this.request.getServerName(),
223                         this.request.getServerPort(),
224                         Logger.format(
225                             "%s%s",
226                             this.request.getContextPath(),
227                             stylesheet
228                         )
229                     ).toString();
230                 } catch (MalformedURLException ex) {
231                     throw new IllegalStateException(ex);
232                 }
233             }
234         } else {
235             stylesheet = ((Stylesheet) antn).value();
236         }
237         return stylesheet;
238     }

240     /**
241      * Configure marhaller adding a xsd validator.
242      * @param mrsh The marshaller, already created and ready to marshal
243      * @param type The class to be marshalled
244      * @see #getContext(Class)
245      */
246     private void addXsdValidatorToMarshaller(final Marshaller mrsh,
247         final Class<?> type) {
248         final String name = XslResolver.schema(type);
249         if (name.isEmpty()) {
250             Logger.debug(
251                 this,
252                 "Schema validation turned off for class '%s'",
253                 type.getName()
254             );
255         } else {
256             final File xsd = new File(this.folder, name);
257             if (xsd.exists()) {
258                 final SchemaFactory factory = SchemaFactory.newInstance(
259                     XMLConstants.W3C_XML_SCHEMA_NS_URI
260                 );
261                 try {
262                     mrsh.setSchema(factory.newSchema(xsd));
263                 } catch (SAXException ex) {
264                     throw new IllegalStateException(
265                         Logger.format(
266                             "Failed to use XSD schema from '%s' for class '%s'",
267                             xsd,
268                             type.getName()
269                         ),
270                         ex
271                     );
272                 }
273                 try {
274                     mrsh.setEventHandler(new XsdEventHandler());
275                 } catch (JAXBException ex) {
276                     throw new IllegalStateException(ex);
277                 }
278                 Logger.debug(
279                     this,
280                     "Class '%s' will be validated with '%s' schema",
281                     type.getName(),
282                     xsd
283                 );
284             } else {
285                 Logger.warn(
286                     this,
287                     "No XSD schema for class '%s' in '%s' file",
288                     type.getName(),
289                     xsd
290                 );
291             }
292         }
293     }
294 
295     /**
296      * Returns the name of XSD schema for this type.
297      * @param type The class
298      * @return The name of XSD file
299      */
300     private static String schema(final Class<?> type) {
301         final Annotation antn = type.getAnnotation(Schema.class);
302         final String schema;
303         if (antn == null) {
304             schema = Logger.format("%s.xsd", type.getName());
305         } else {
306             if (((Schema) antn).ignore()) {
307                 schema = "";
308             } else {
309                 schema = ((Schema) antn).value();
310             }
311         }
312         return schema;
313     }
