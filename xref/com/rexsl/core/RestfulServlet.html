56  /**
57   * The only and the main servlet from ReXSL framework.
58   *
59   * <p>You don't need to instantiate this class directly. It is instantiated
60   * by servlet container according to configuration from {@code web.xml}.
61   * Should be used in {@code web.xml} (together with {@link XsltFilter})
62   * like that:
63   *
64   * <pre> &lt;servlet>
65   *  &lt;servlet-name>RestfulServlet&lt;/servlet-name>
66   *  &lt;servlet-class>com.rexsl.core.RestfulServlet&lt;/servlet-class>
67   *  &lt;init-param>
68   *   &lt;param-name>com.rexsl.PACKAGES&lt;/param-name>
69   *   &lt;param-value>com.rexsl.foo&lt;/param-value>
70   *  &lt;/init-param>
71   * &lt;/servlet>
72   * &lt;servlet-mapping>
73   *  &lt;servlet-name>RestfulServlet&lt;/servlet-name>
74   *  &lt;url-pattern>/*&lt;/url-pattern>
75   * &lt;/servlet-mapping></pre>
76   *
77   * <p>{@code com.rexsl.PACKAGES} init parameter should contain comma-separated
78   * list of packages where JAX-RS annotated resources are located and should be
79   * discovered. If this parameter is not set a runtime exception will be thrown
80   * and the servlet won't be initialized. The same will happen if the parameter
81   * contains incorrect data. We will consider a package is valid if and only if
82   * it abides to the Java package naming conventions.
83   *
84   * @author Yegor Bugayenko (yegor@tpc2.com)
85   * @version $Id$
86   * @see <a href="http://www.rexsl.com">Introduction to ReXSL</a>
87   * @see <a href="http://www.oracle.com/technetwork/java/javaee/servlet/index.html">Java Servlet Technology</a>
88   * @see <a href="http://docs.oracle.com/javase/specs/jls/se7/html/jls-6.html#jls-6.1">jls-6.1</a>
89   * @see <a href="http://www.oracle.com/technetwork/java/codeconventions-135099.html">Java naming conventions</a>
90   * @see <a href="http://docs.oracle.com/javase/tutorial/java/package/namingpkgs.html">Naming a package</a>
91   * @since 0.2
92   */
93  @ToString
94  @EqualsAndHashCode(callSuper = true)
95  @Loggable(Loggable.DEBUG)
96  public final class RestfulServlet extends ServletContainer {

98      /**
99       * Name of servlet init param.
100      *
101      * <p>This parameter should be used in servlet initialization section of
102      * {@code web.xml} in order to tell the servlet in which packages your
103      * JAX-RS annotated resources are located. The value of the parameter
104      * should contain a comma-separated list of Java package names.
105      *
106      * @see RestfulServlet
107      * @since 0.3.7
108      */
109     public static final String PACKAGES = "com.rexsl.PACKAGES";

116     /**
117      * Comma, a separator between package names.
118      */
119     private static final String COMMA = ",";

121     /**
122      * {@inheritDoc}
123      * @checkstyle RedundantThrows (4 lines)
124      */
125     @Override
126     public void init(@NotNull final ServletConfig config)
127         throws ServletException {
128         final Collection<String> packages = new HashSet<String>(0);
129         packages.add(this.getClass().getPackage().getName());
130         final String param = config.getInitParameter(RestfulServlet.PACKAGES);
131         if (param == null) {
132             throw new ServletException(
133                 Logger.format(
134                     "'%s' servlet parameter is mandatory",
135                     RestfulServlet.PACKAGES
136                 )
137             );
138         }
139         for (final String name
140             : StringUtils.split(param, RestfulServlet.COMMA)) {
141             final String pkg = name.trim();
142             final Pattern ptrn = Pattern.compile(
143                 "^([a-z_][a-z0-9_]*(\\.[a-z_][a-z0-9_]*)*)$"
144             );
145             final Matcher match = ptrn.matcher(pkg);
146             if (!match.matches()) {
147                 throw new ServletException(
148                     Logger.format(
149                         // @checkstyle LineLength (1 line)
150                         "'%s' servlet parameter contains non-valid data: %s",
151                         RestfulServlet.PACKAGES,
152                         pkg
153                     )
154                 );
155             }
156             packages.add(pkg);
157             Logger.info(
158                 this,
159                 "#init(): '%s' package added (%d total)",
160                 pkg,
161                 packages.size()
162             );
163         }
164         final Properties props = new Properties();
165         props.setProperty(
166             PackagesResourceConfig.PROPERTY_PACKAGES,
167             StringUtils.join(packages, RestfulServlet.COMMA)
168         );
169         this.reconfigureJUL();
170         this.init(new ServletConfigWrapper(config, props));
171     }

173     /**
174      * {@inheritDoc}
175      * @checkstyle ThrowsCount (6 lines)
176      * @checkstyle RedundantThrows (5 lines)
177      */
178     @Override
179     public void service(final HttpServletRequest request,
180         final HttpServletResponse response)
181         throws ServletException, IOException {
182         final long start = System.currentTimeMillis();
183         super.service(request, response);
184         final long duration = System.currentTimeMillis() - start;
185         if (duration > TimeUnit.SECONDS.toMillis(1L)) {
186             Logger.warn(
187                 this,
188                 "#service(%s %s): %[ms]s is too slow (IP=%s)",
189                 request.getMethod(),
190                 request.getRequestURI(),
191                 duration,
192                 request.getRemoteAddr()
193             );
194         }
195         response.addHeader("X-Rexsl-Millis", Long.toString(duration));
196         response.addHeader(
197             "X-Rexsl-Version",
198             String.format(
199                 "%s/r%s",
200                 Manifests.read("ReXSL-Version"),
201                 Manifests.read("ReXSL-Build")
202             )
203         );
204     }

206     /**
207      * Initialize JUL-to-SLF4J bridge.
208      * @see #init(ServletConfig)
209      */
210     private void reconfigureJUL() {
211         final java.util.logging.Logger root =
212             LogManager.getLogManager().getLogger("");
213         final Handler[] handlers = root.getHandlers();
214         for (final Handler handler : handlers) {
215             root.removeHandler(handler);
216         }
217         SLF4JBridgeHandler.install();
218         Logger.debug(this, "#julToSlf4j(): JUL forwarded to SLF4j");
219     }
