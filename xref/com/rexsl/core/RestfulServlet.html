58  /**
59   * The only and the main servlet from ReXSL framework.
60   *
61   * <p>You don't need to instantiate this class directly. It is instantiated
62   * by servlet container according to configuration from {@code web.xml}.
63   * Should be used in {@code web.xml} (together with {@link XsltFilter})
64   * like that:
65   *
66   * <pre> &lt;servlet>
67   *  &lt;servlet-name>RestfulServlet&lt;/servlet-name>
68   *  &lt;servlet-class>com.rexsl.core.RestfulServlet&lt;/servlet-class>
69   *  &lt;init-param>
70   *   &lt;param-name>com.rexsl.PACKAGES&lt;/param-name>
71   *   &lt;param-value>com.rexsl.foo&lt;/param-value>
72   *  &lt;/init-param>
73   * &lt;/servlet>
74   * &lt;servlet-mapping>
75   *  &lt;servlet-name>RestfulServlet&lt;/servlet-name>
76   *  &lt;url-pattern>/*&lt;/url-pattern>
77   * &lt;/servlet-mapping></pre>
78   *
79   * <p>{@code com.rexsl.PACKAGES} init parameter should contain comma-separated
80   * list of packages where JAX-RS annotated resources are located and should be
81   * discovered. If this parameter is not set a runtime exception will be thrown
82   * and the servlet won't be initialized. The same will happen if the parameter
83   * contains incorrect data. We will consider a package is valid if and only if
84   * it abides to the Java package naming conventions.
85   *
86   * @author Yegor Bugayenko (yegor@tpc2.com)
87   * @version $Id$
88   * @see <a href="http://www.rexsl.com">Introduction to ReXSL</a>
89   * @see <a href="http://www.oracle.com/technetwork/java/javaee/servlet/index.html">Java Servlet Technology</a>
90   * @see <a href="http://docs.oracle.com/javase/specs/jls/se7/html/jls-6.html#jls-6.1">jls-6.1</a>
91   * @see <a href="http://www.oracle.com/technetwork/java/codeconventions-135099.html">Java naming conventions</a>
92   * @see <a href="http://docs.oracle.com/javase/tutorial/java/package/namingpkgs.html">Naming a package</a>
93   * @since 0.2
94   */
95  @ToString
96  @EqualsAndHashCode(callSuper = true)
97  @Loggable(Loggable.DEBUG)
98  public final class RestfulServlet extends ServletContainer {
99  
100     /**
101      * Name of servlet init param.
102      *
103      * <p>This parameter should be used in servlet initialization section of
104      * {@code web.xml} in order to tell the servlet in which packages your
105      * JAX-RS annotated resources are located. The value of the parameter
106      * should contain a comma-separated list of Java package names.
107      *
108      * @see RestfulServlet
109      * @since 0.3.7
110      */
111     public static final String PACKAGES = "com.rexsl.PACKAGES";
112 
113     /**
114      * Serialization marker.
115      */
116     private static final long serialVersionUID = 0x7526FA78EED21470L;
117 
118     /**
119      * Comma, a separator between package names.
120      */
121     private static final String COMMA = ",";
122 
123     /**
124      * {@inheritDoc}
125      * @checkstyle RedundantThrows (4 lines)
126      */
127     @Override
128     public void init(@NotNull final ServletConfig config)
129         throws ServletException {
130         final Collection<String> packages = new HashSet<String>(0);
131         packages.add(this.getClass().getPackage().getName());
132         final String param = config.getInitParameter(RestfulServlet.PACKAGES);
133         if (param == null) {
134             throw new ServletException(
135                 Logger.format(
136                     "'%s' servlet parameter is mandatory",
137                     RestfulServlet.PACKAGES
138                 )
139             );
140         }
141         for (final String name
142             : StringUtils.split(param, RestfulServlet.COMMA)) {
143             final String pkg = name.trim();
144             final Pattern ptrn = Pattern.compile(
145                 "^([a-z_][a-z0-9_]*(\\.[a-z_][a-z0-9_]*)*)$"
146             );
147             final Matcher match = ptrn.matcher(pkg);
148             if (!match.matches()) {
149                 throw new ServletException(
150                     Logger.format(
151                         // @checkstyle LineLength (1 line)
152                         "'%s' servlet parameter contains non-valid data: %s",
153                         RestfulServlet.PACKAGES,
154                         pkg
155                     )
156                 );
157             }
158             packages.add(pkg);
159             Logger.info(
160                 this,
161                 "#init(): '%s' package added (%d total)",
162                 pkg,
163                 packages.size()
164             );
165         }
166         final Properties props = new Properties();
167         props.setProperty(
168             PackagesResourceConfig.PROPERTY_PACKAGES,
169             StringUtils.join(packages, RestfulServlet.COMMA)
170         );
171         this.reconfigureJUL();
172         this.init(new ServletConfigWrapper(config, props));
173     }
174 
175     /**
176      * {@inheritDoc}
177      * @checkstyle ThrowsCount (8 lines)
178      * @checkstyle RedundantThrows (8 lines)
179      */
180     @Override
181     @Loggable(value = Loggable.DEBUG, ignore = WebApplicationException.class)
182     public void service(final HttpServletRequest request,
183         final HttpServletResponse response)
184         throws ServletException, IOException {
185         final long start = System.currentTimeMillis();
186         super.service(request, response);
187         final long duration = System.currentTimeMillis() - start;
188         if (duration > TimeUnit.SECONDS.toMillis(1L)) {
189             Logger.warn(
190                 this,
191                 "#service(%s %s): %[ms]s is too slow (IP=%s)",
192                 request.getMethod(),
193                 request.getRequestURI(),
194                 duration,
195                 request.getRemoteAddr()
196             );
197         }
198         response.addHeader("X-Rexsl-Millis", Long.toString(duration));
199         response.addHeader(
200             "X-Rexsl-Version",
201             String.format(
202                 "%s/r%s",
203                 Manifests.read("ReXSL-Version"),
204                 Manifests.read("ReXSL-Build")
205             )
206         );
207     }
208 
209     /**
210      * Initialize JUL-to-SLF4J bridge.
211      * @see #init(ServletConfig)
212      */
213     private void reconfigureJUL() {
214         final LevelChangePropagator listener = new LevelChangePropagator();
215         final LoggerContext loggerContext = (LoggerContext) LoggerFactory
216             .getILoggerFactory();
217         loggerContext.addListener(listener);
218         listener.setContext(loggerContext);
219         loggerContext.start();
220         SLF4JBridgeHandler.removeHandlersForRootLogger();
221         SLF4JBridgeHandler.install();
222         Logger.debug(this, "#julToSlf4j(): JUL forwarded to SLF4j");
223     }
