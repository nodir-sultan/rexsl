54  /**
55   * Flash message (through cookie).
56   *
57   * @author Yegor Bugayenko (yegor@tpc2.com)
58   * @version $Id$
59   * @since 0.4.8
60   * @see BasePage
61   * @see <a href="http://www.rexsl.com/rexsl-page/inset-flash.html">Flash messages in RESTful interfaces</a>
62   */
63  @ToString
64  @EqualsAndHashCode(of = "resource")
65  @Loggable(Loggable.DEBUG)
66  @SuppressWarnings("PMD.TooManyMethods")
67  public final class FlashInset implements Inset {
68  
69      /**
70       * ImmutableHeader name.
71       */
72      private static final String HEADER = "X-Rexsl-Flash";
73  
74      /**
75       * Cookie name.
76       */
77      private static final String COOKIE = "Rexsl-Flash";
78  
79      /**
80       * The resource.
81       */
82      private final transient Resource resource;
83  
84      /**
85       * Public ctor.
86       * @param res The resource
87       */
88      public FlashInset(@NotNull final Resource res) {
89          this.resource = res;
90      }
91  
92      @Override
93      public void render(@NotNull final BasePage<?, ?> page,
94          @NotNull final Response.ResponseBuilder builder) {
95          if (this.resource.httpHeaders().getCookies()
96              .containsKey(FlashInset.COOKIE)) {
97              final FlashInset.Flash cookie = new FlashInset.Flash(
98                  this.resource.httpHeaders().getCookies().get(FlashInset.COOKIE)
99              );
100             page.append(
101                 FlashInset.bundle(
102                     cookie.level(),
103                     cookie.message(),
104                     cookie.msec()
105                 )
106             );
107             builder.cookie(
108                 new CookieBuilder(this.resource.uriInfo().getBaseUri())
109                     .name(FlashInset.COOKIE)
110                     .build()
111             );
112         }
113     }

115     /**
116      * Make a bundle.
117      * @param level Logging level
118      * @param msg Message
119      * @param msec Time spent
120      * @return JaxbBundle injectable into the page
121      */
122     public static JaxbBundle bundle(@NotNull final Level level,
123         @NotNull final String msg, final long msec) {
124         return new JaxbBundle("flash")
125             .add("message", msg).up()
126             .add("level", level.toString()).up()
127             .add("msec", Long.toString(msec)).up();
128     }

137     public static JaxbBundle bundle(final Level level, final String msg) {
138         return FlashInset.bundle(level, msg, 0L);

141     /**
142      * Create an exception that will redirect to the page with an error message.
143      *
144      * <p>The difference between this method and a static
145      * {@link #forward(URI,String,Level)} is in its awareness of a resource,
146      * which is forwarding. Key benefit is that a non-static method adds extra
147      * value to the cookie, which is time consumed by the resource until the
148      * redirect happened.
149      *
150      * @param uri The URI to forward to
151      * @param message The message to show as error
152      * @param level Message level
153      * @return The exception to throw
154      */
155     public WebApplicationException redirect(@NotNull final URI uri,
156         @NotNull final String message, @NotNull final Level level) {
157         return new FlashException(
158             Response.status(HttpURLConnection.HTTP_SEE_OTHER)
159                 .location(uri)
160                 .cookie(
161                     new FlashInset.Flash(
162                         uri, message, level,
163                         System.currentTimeMillis() - this.resource.started()
164                     )
165                 )
166                 .header(FlashInset.HEADER, message)
167                 .entity(message)
168                 .build(),
169             message,
170             level
171         );
172     }

174     /**
175      * Create an exception that will redirect to the page with an error message.
176      *
177      * <p>The difference between this method and a static
178      * {@link #forward(URI,Exception)} is in its awareness of a resource,
179      * which is forwarding. Key benefit is that a non-static method adds extra
180      * value to the cookie, which is time consumed by the resource until the
181      * redirect happened.
182      *
183      * @param uri The URI to forward to
184      * @param cause The cause of this problem
185      * @return The exception to throw
186      */
187     public WebApplicationException redirect(@NotNull final URI uri,
188         @NotNull final Exception cause) {
189         return this.redirect(uri, cause.getMessage(), Level.SEVERE);
190     }

224     /**
225      * The cookie.
226      */
227     public static final class Flash extends NewCookie {
228         /**
229          * Total elements expected in a cookie text.
230          */
231         @SuppressWarnings("PMD.UnusedPrivateField")
232         private static final int TOTAL = 3;
233         /**
234          * The message.
235          */
236         private final transient String msg;
237         /**
238          * Level of message.
239          */
240         private final transient Level lvl;
241         /**
242          * Milliseconds consumed (or -1).
243          */
244         private final transient long millis;
245         /**
246          * Public ctor, from a cookie encoded text.
247          * @param cookie The cookie
248          */
249         public Flash(@NotNull final Cookie cookie) {
250             super(FlashInset.COOKIE, cookie.getValue());
251             if (StringUtils.isBlank(cookie.getValue())) {
252                 throw new WebApplicationException(
253                     Response.status(HttpURLConnection.HTTP_BAD_REQUEST)
254                         .cookie(new NewCookie(cookie, "", 0, false))
255                         .build()
256                 );
257             }
258             final String decoded = FlashInset.Flash.decode(this.getValue());
259             final String[] parts = decoded.split(":", FlashInset.Flash.TOTAL);
260             Validate.isTrue(
261                 parts.length == FlashInset.Flash.TOTAL,
262                 // @checkstyle LineLength (1 line)
263                 "can't decode cookie '%s' (decoded='%s'), %d parts expected but %d found",
264                 this.getValue(),
265                 decoded,
266                 FlashInset.Flash.TOTAL,
267                 parts.length
268             );
269             Validate.isTrue(
270                 parts[0].matches("INFO|WARNING|SEVERE"),
271                 "can't detect level of cookie '%s' (decoded='%s')",
272                 this.getValue(),
273                 decoded
274             );
275             this.lvl = Level.parse(parts[0]);
276             Validate.isTrue(
277                 parts[1].matches("-1|\\d+"),
278                 "can't parse milliseconds in cookie '%s' (decoded='%s')",
279                 this.getValue(),
280                 decoded
281             );
282             this.millis = Long.parseLong(parts[1]);
283             Validate.notBlank(
284                 parts[2],
285                 "empty message in cookie '%s' (decoded='%s')",
286                 this.getValue(),
287                 decoded
288             );
289             this.msg = parts[2];
290         }
291         /**
292          * Public ctor, from exact values.
293          * @param base Base URI where we're using it
294          * @param message The message
295          * @param level The level
296          */
297         public Flash(@NotNull final URI base,
298             @NotNull final String message, @NotNull final Level level) {
299             this(base, message, level, -1);
300         }
301         /**
302          * Public ctor, from exact values.
303          * @param base Base URI where we're using it
304          * @param message The message
305          * @param level The level
306          * @param msec Milliseconds consumed
307          * @checkstyle ParameterNumber (5 lines)
308          */
309         public Flash(@NotNull final URI base,
310             @NotNull final String message, @NotNull final Level level,
311             final long msec) {
312             super(
313                 new CookieBuilder(base)
314                     .name(FlashInset.COOKIE)
315                     .value(FlashInset.Flash.encode(message, level, msec))
316                     .temporary()
317                     .build()
318             );
319             Validate.notBlank(message, "flash message can't be empty");
320             this.msg = message;
321             this.lvl = level;
322             Validate.isTrue(
323                 msec >= 0L || msec == -1L,
324                 "milliseconds can either be positive or equal to -1"
325             );
326             this.millis = msec;
327         }
328         /**
329          * Get message.
330          * @return The message
331          */
332         public String message() {
333             return this.msg;
334         }
335         /**
336          * Get color of it.
337          * @return The color
338          */
339         public Level level() {
340             return this.lvl;
341         }
342         /**
343          * Milliseconds consumed.
344          * @return The amount of them
345          */
346         public long msec() {
347             return this.millis;
348         }
349         @Override
350         public String getPath() {
351             return "/";
352         }
353         /**
354          * Encode message and color.
355          * @param message The message
356          * @param level The level
357          * @param msec Milliseconds consumed
358          * @return Encoded cookie value
359          */
360         private static String encode(final String message, final Level level,
361             final long msec) {
362             try {
363                 return Base64.encodeBase64URLSafeString(
364                     String.format("%s:%d:%s", level, msec, message)
365                         .getBytes(CharEncoding.UTF_8)
366                 );
367             } catch (final UnsupportedEncodingException ex) {
368                 throw new IllegalStateException(ex);
369             }
370         }
371         /**
372          * Decode text.
373          * @param text The text to decode (cookie value)
374          * @return Decoded value
375          */
376         private static String decode(final String text) {
377             try {
378                 return new String(
379                     Base64.decodeBase64(text),
380                     CharEncoding.UTF_8
381                 );
382             } catch (final UnsupportedEncodingException ex) {
383                 throw new IllegalStateException(ex);
384             }
385         }
386 
387     }
