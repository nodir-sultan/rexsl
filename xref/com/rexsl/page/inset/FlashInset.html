53  /**
54   * Flash message (through cookie).
55   *
56   * @author Yegor Bugayenko (yegor@tpc2.com)
57   * @version $Id$
58   * @since 0.4.8
59   * @see BasePage
60   * @see <a href="http://www.rexsl.com/rexsl-page/inset-flash.html">Flash messages in RESTful interfaces</a>
61   */
62  @ToString
63  @EqualsAndHashCode(of = "resource")
64  @Loggable(Loggable.DEBUG)
65  @SuppressWarnings("PMD.TooManyMethods")
66  public final class FlashInset implements Inset {
67  
68      /**
69       * ImmutableHeader name.
70       */
71      private static final String HEADER = "X-Rexsl-Flash";
72  
73      /**
74       * Cookie name.
75       */
76      private static final String COOKIE = "Rexsl-Flash";

78      /**
79       * The resource.
80       */
81      private final transient Resource resource;

83      /**
84       * Public ctor.
85       * @param res The resource
86       */
87      public FlashInset(@NotNull final Resource res) {
88          this.resource = res;
89      }
90  
91      @Override
92      public void render(@NotNull final BasePage<?, ?> page,
93          @NotNull final Response.ResponseBuilder builder) {
94          if (this.resource.httpHeaders().getCookies()
95              .containsKey(FlashInset.COOKIE)) {
96              final FlashInset.Flash cookie = new FlashInset.Flash(
97                  this.resource.httpHeaders().getCookies().get(FlashInset.COOKIE)
98              );
99              page.append(
100                 FlashInset.bundle(
101                     cookie.level(),
102                     cookie.message(),
103                     cookie.msec()
104                 )
105             );
106             builder.cookie(
107                 new CookieBuilder(this.resource.uriInfo().getBaseUri())
108                     .name(FlashInset.COOKIE)
109                     .build()
110             );
111         }
112     }

117      * @param msg Message

121     public static JaxbBundle bundle(@NotNull final Level level,
122         @NotNull final String msg, final long msec) {
123         return new JaxbBundle("flash")
124             .add("message", msg).up()

126             .add("msec", Long.toString(msec)).up();

140     /**
141      * Create an exception that will redirect to the page with an error message.
142      *
143      * <p>The difference between this method and a static
144      * {@link #forward(URI,String,Level)} is in its awareness of a resource,
145      * which is forwarding. Key benefit is that a non-static method adds extra
146      * value to the cookie, which is time consumed by the resource until the
147      * redirect happened.
148      *
149      * @param uri The URI to forward to
150      * @param message The message to show as error
151      * @param level Message level
152      * @return The exception to throw
153      */
154     public WebApplicationException redirect(@NotNull final URI uri,
155         @NotNull final String message, @NotNull final Level level) {
156         return new FlashException(
157             Response.status(HttpURLConnection.HTTP_SEE_OTHER)
158                 .location(uri)
159                 .cookie(
160                     new FlashInset.Flash(
161                         uri, message, level,
162                         System.currentTimeMillis() - this.resource.started()
163                     )
164                 )
165                 .header(FlashInset.HEADER, message)
166                 .entity(message)
167                 .build(),
168             message,
169             level
170         );
171     }

223     /**
224      * The cookie.
225      */
226     public static final class Flash extends NewCookie {
227         /**
228          * Total elements expected in a cookie text.
229          */
230         private static final int TOTAL = 3;
231         /**
232          * The message.
233          */
234         private final transient String msg;
235         /**
236          * Level of message.
237          */
238         private final transient Level lvl;
239         /**
240          * Milliseconds consumed (or -1).
241          */
242         private final transient long millis;
243         /**
244          * Public ctor, from a cookie encoded text.
245          * @param cookie The cookie
246          */
247         public Flash(@NotNull final Cookie cookie) {
248             super(FlashInset.COOKIE, cookie.getValue());
249             Validate.notBlank(cookie.getValue(), "cookie value is blank");
250             final String decoded = FlashInset.Flash.decode(this.getValue());
251             final String[] parts = decoded.split(":", FlashInset.Flash.TOTAL);
252             Validate.isTrue(
253                 parts.length == FlashInset.Flash.TOTAL,
254                 // @checkstyle LineLength (1 line)
255                 "can't decode cookie '%s' (decoded='%s'), %d parts expected but %d found",
256                 this.getValue(),
257                 decoded,
258                 FlashInset.Flash.TOTAL,
259                 parts.length
260             );
261             Validate.isTrue(
262                 parts[0].matches("INFO|WARNING|SEVERE"),
263                 "can't detect level of cookie '%s' (decoded='%s')",
264                 this.getValue(),
265                 decoded
266             );
267             this.lvl = Level.parse(parts[0]);
268             Validate.isTrue(
269                 parts[1].matches("-1|\\d+"),
270                 "can't parse milliseconds in cookie '%s' (decoded='%s')",
271                 this.getValue(),
272                 decoded
273             );
274             this.millis = Long.parseLong(parts[1]);
275             Validate.notBlank(
276                 parts[2],
277                 "empty message in cookie '%s' (decoded='%s')",
278                 this.getValue(),
279                 decoded
280             );
281             this.msg = parts[2];
282         }
283         /**
284          * Public ctor, from exact values.
285          * @param base Base URI where we're using it
286          * @param message The message
287          * @param level The level
288          */
289         public Flash(@NotNull final URI base,
290             @NotNull final String message, @NotNull final Level level) {
291             this(base, message, level, -1);
292         }
293         /**
294          * Public ctor, from exact values.
295          * @param base Base URI where we're using it
296          * @param message The message
297          * @param level The level
298          * @param msec Milliseconds consumed
299          * @checkstyle ParameterNumber (5 lines)
300          */
301         public Flash(@NotNull final URI base,
302             @NotNull final String message, @NotNull final Level level,
303             final long msec) {
304             super(
305                 new CookieBuilder(base)
306                     .name(FlashInset.COOKIE)
307                     .value(FlashInset.Flash.encode(message, level, msec))
308                     .temporary()
309                     .build()
310             );
311             Validate.notBlank(message, "flash message can't be empty");
312             this.msg = message;
313             this.lvl = level;
314             Validate.isTrue(
315                 msec >= 0L || msec == -1L,
316                 "milliseconds can either be positive or equal to -1"
317             );
318             this.millis = msec;
319         }
320         /**
321          * Get message.
322          * @return The message
323          */
324         public String message() {
325             return this.msg;
326         }
327         /**
328          * Get color of it.
329          * @return The color
330          */
331         public Level level() {
332             return this.lvl;
333         }
334         /**
335          * Milliseconds consumed.
336          * @return The amount of them
337          */
338         public long msec() {
339             return this.millis;
340         }
341         @Override
342         public String getPath() {
343             return "/";
344         }
345         /**
346          * Encode message and color.
347          * @param message The message
348          * @param level The level
349          * @param msec Milliseconds consumed
350          * @return Encoded cookie value
351          */
352         private static String encode(final String message, final Level level,
353             final long msec) {
354             try {
355                 return Base64.encodeBase64URLSafeString(
356                     String.format("%s:%d:%s", level, msec, message)
357                         .getBytes(CharEncoding.UTF_8)
358                 );
359             } catch (UnsupportedEncodingException ex) {
360                 throw new IllegalStateException(ex);
361             }
362         }
363         /**
364          * Decode text.
365          * @param text The text to decode (cookie value)
366          * @return Decoded value
367          */
368         private static String decode(final String text) {
369             try {
370                 return new String(
371                     Base64.decodeBase64(text),
372                     CharEncoding.UTF_8
373                 );
374             } catch (UnsupportedEncodingException ex) {
375                 throw new IllegalStateException(ex);
376             }
377         }
378 
379     }
