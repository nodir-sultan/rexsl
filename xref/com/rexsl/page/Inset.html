32  import java.lang.annotation.Documented;
33  import java.lang.annotation.ElementType;
34  import java.lang.annotation.Retention;
35  import java.lang.annotation.RetentionPolicy;
36  import java.lang.annotation.Target;
37  import javax.ws.rs.core.Response;

39  /**
40   * Insertion into a page.
41   *
42   * <p>Post-rendering strategy applied to a {@link BasePage} and
43   * {@link Response.Builder}. You can define your own insets as anonymous
44   * classes, for example:
45   *
46   * <pre> &#64;Inset.Default({ LinksInset.class, FlashInset.class })
47   * public class BaseRs extends BaseResource {
48   *   &#64;Inset.Runtime
49   *   public Inset ver() {
50   *     return new VersionInset("1.0", "", "13-Mar-2013");
51   *   }
52   *   &#64;Inset.Runtime
53   *   public Inset supplementary() {
54   *     return new Inset() {
55   *       &#64;Override
56   *       public void render(final BasePage&lt;?, ?&gt; page,
57   *         final Response.ResponseBuilder builder) {
58   *         builder.type(MediaType.TEXT_XML);
59   *         builder.header(HttpHeaders.VARY, "Cookie");
60   *     }
61   *   };
62   * }</pre>
63   *
64   * <p>For every new page methods {@code version()} and {@code supplementary()}
65   * will be called. Returned insets will be used to help in page rendering.
66   * Their {@code render(BasePage,Response.ResponseBuilder)} methods will be
67   * used to extend the JAXB page and the JAX-RS response.
68   *
69   * @author Yegor Bugayenko (yegor@tpc2.com)
70   * @version $Id$
71   * @since 0.4.8
72   * @see BasePage
73   */
74  public interface Inset {
75  
76      /**
77       * Annotates a method of JAX-RS resource that returns an instance
78       * of {@code Inset}.
79       *
80       * <p>The method will be called automatically for every page rendered. The
81       * returned {@code Inset} will be used to render the page. If an annotated
82       * method returns an object, which is not a instance of {@code Inset} a
83       * runtime exception will be thrown.
84       */
85      @Documented
86      @Retention(RetentionPolicy.RUNTIME)
87      @Target(ElementType.METHOD)
88      @interface Runtime {
89      }
90  
91      /**
92       * Annotates a JAX-RS resource, informing the rendering mechanism about
93       * all Inset classes required for page rendering.
94       *
95       * <p>All classes listed in the annotation will be instantiated with
96       * one-argument constructors. If such constructor is absent a runtime
97       * exception is thrown.
98       */
99      @Documented
100     @Retention(RetentionPolicy.RUNTIME)
101     @Target(ElementType.TYPE)
102     @interface Default {
103         /**
104          * Class to instantiate.
105          */
106         Class<? extends Inset>[] value();
107     }
108 
109     /**
110      * Render it into the response builder.
111      * @param page The page to render
112      * @param builder The builder
113      */
114     void render(BasePage<?, ?> page, Response.ResponseBuilder builder);
115 
116 }
