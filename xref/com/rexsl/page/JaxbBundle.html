48  /**
49   * JAXB bundle.
50   *
51   * <p>It's a convenient instrument that enables on-fly creation of DOM/XML
52   * structures, for example
53   * (<a href="http://en.wikipedia.org/wiki/Fluent_interface">fluent
54   * interface</a>):
55   *
56   * <pre> Element element = new JaxbBundle("root")
57   *   .add("employee")
58   *     .attr("age", "28")
59   *     .add("dept", "Software")
60   *       .attr("country", "DE")
61   *     .up()
62   *     .add("salary", "> &#92;u20AC 50,000")
63   *     .up()
64   *     .add("rank", "high")
65   *   .up()
66   *   .attr("time", new Date())
67   *   .element();</pre>
68   *
69   * <p>If you convert this {@code element} to XML this is how it will look:
70   *
71   * <pre> &lt;?xml version="1.0" ?&gt;
72   * &lt;root time="Sun Jul 20 16:17:00 EDT 1969"&gt;
73   *   &lt;employee age="28"&gt;
74   *     &lt;dept country="DE"&gt;Software&lt;/dept&gt;
75   *     &lt;salary&gt;&amp;gt; &#x20AC; 50,000&lt;/salary&gt;
76   *     &lt;rank&gt;high&lt;/rank&gt;
77   *   &lt;/employee&gt;
78   * &lt;/root&gt;</pre>
79   *
80   * <p>Then, you can add this {@link Element} to your JAXB object, and return
81   * it from a method annotated with {@link javax.xml.bind.annotation.XmlElement},
82   * for example:
83   *
84   * <pre> &#64;XmlRootElement
85   * public class Page {
86   *   &#64;XmlElement
87   *   public Object getEmployee() {
88   *     return new JaxbBundle("employee")
89   *       .attr("age", "35")
90   *       .attr("country", "DE")
91   *       .add("salary", "> &#92;u20AC 50,000")
92   *       .up()
93   *       .element();
94   *   }
95   * }</pre>
96   *
97   * <p>This mechanism, very often, is much more convenient and shorter than
98   * a declaration of a new POJO every time you need to return a small piece
99   * of XML data.
100  *
101  * <p>Since version 0.4.10 it's possible to add links, groups, and other
102  * bundles. For example:
103  *
104  * <pre> final Element element = new JaxbBundle("garage")
105  *   .link(new Link("add", "/add-car"))
106  *   .add("cars").add(
107  *     new JaxbBundle.Group&lt;String&gt;(cars) {
108  *       &#64;Override
109  *       public JaxbBundle bundle(Car car) {
110  *         return new JaxbBundle("car").add("make", car.make()).up();
111  *       }
112  *     }
113  *   ).up()
114  *   .add(new JaxbBundle("owner").add("email", "...").up())
115  *   .element();</pre>
116  *
117  * <p>The class is mutable and thread-safe.
118  *
119  * @todo #430 A new design would be required for this class. <br/>
120  *  <b>The problems</b>:
121  *  <ul>
122  *  <li>this.content and this.parent are used as flags, not only as data
123  *  storage variables.
124  *  <li>there are three different "types" combined in
125  *  one class: root element (parent == null), node element (content ==
126  *  null), and leaf element.
127  *  </ul>
128  *  <br/>
129  *  <b>The solution</b>: A new hierarchy of classes: <br/>
130  *  <code>
131  *  interface TreeNode<br/>
132  *  <br/>
133  *  abstract AbstractNode implements TreeNode<br/>
134  *  +name: String<br/>
135  *  +attributes: ConcurrentMap<String, String><br/>
136  *  <br/>
137  *  abstract SubNode extends AbstractNode<br/>
138  *  +parent: TreeNode<br/>
139  *  <br/>
140  *  InnerNode extends SubNode<br/>
141  *  +children: List<TreeNode><br/>
142  *  <br/>
143  *  LeafNode extends SubNode<br/>
144  *  +content: String<br/>
145  *  <br/>
146  *  DeadNode implements TreeNode<br/>
147  *  // this is the node we set as parent in root node<br/>
148  *  </code> <br/>
149  * @author Yegor Bugayenko (yegor@tpc2.com)
150  * @version $Id$
151  * @since 0.3.7
152  */
153 @SuppressWarnings({ "PMD.NullAssignment", "PMD.TooManyMethods" })
154 @ToString(of = { "name", "content" })
155 @EqualsAndHashCode(of = { "name", "content", "parent" })
156 @Loggable(Loggable.DEBUG)
157 public final class JaxbBundle {
158 
159     /**
160      * Parent bundle, if exists.
161      */
162     private final transient JaxbBundle parent;
163 
164     /**
165      * Name of it.
166      */
167     @NotNull
168     private final transient String name;
169 
170     /**
171      * Text content of it.
172      */
173     private final transient String content;
174 
175     /**
176      * Children.
177      */
178     private final transient List<JaxbBundle> children =
179         new CopyOnWriteArrayList<JaxbBundle>();
180 
181     /**
182      * DOM children.
183      */
184     private final transient List<Element> elements =
185         new CopyOnWriteArrayList<Element>();
186 
187     /**
188      * Links.
189      */
190     private final transient List<Link> links =
191         new CopyOnWriteArrayList<Link>();
192 
193     /**
194      * Attributes.
195      */
196     private final transient ConcurrentMap<String, String> attrs =
197         new ConcurrentHashMap<String, String>(0);
198 
199     /**
200      * Default ctor, for JAXB (always throws a runtime exception).
201      *
202      * <p>You're not supposed to use this constructor. Instead, use either
203      * {@link JaxbBundle(String)} or {@link JaxbBundle(String,String)}.
204      */
205     public JaxbBundle() {
206         throw new IllegalStateException(
207             "JaxbBundle() ctor can't be used, use JaxbBundle(String) instead"
208         );
209     }
210 
211     /**
212      * Public ctor, with just a name of XML element and no text content.
213      * @param nam The name of XML element
214      */
215     public JaxbBundle(@NotNull final String nam) {
216         this.parent = null;
217         this.name = nam;
218         this.content = "";
219     }
220 
221     /**
222      * Public ctor, with XML element name and its content.
223      * @param nam The name of XML element
224      * @param text Text content of the XML document
225      */
226     public JaxbBundle(@NotNull final String nam, @NotNull final String text) {
227         this.parent = null;
228         this.name = nam;
229         this.content = text;
230     }
231 
232     /**
233      * Public ctor.
234      * @param prnt Not-null parent bundle
235      * @param nam The name of XML element
236      * @param text Text content of the XML element
237      */
238     private JaxbBundle(@NotNull final JaxbBundle prnt,
239         @NotNull final String nam, @NotNull final String text) {
240         this.parent = prnt;
241         this.name = nam;
242         this.content = text;
243     }
244 
245     /**
246      * Add new child XML element.
247      * @param nam The name of child element
248      * @return The child bundle (use {@link #up()} on it in order to get back to
249      *  the parent bundle)
250      */
251     @NotNull
252     public JaxbBundle add(@NotNull final String nam) {
253         return this.add(nam, "");
254     }
255 
256     /**
257      * Add new child XML element.
258      * @param element The DOM element to add
259      * @return The same bundle
260      * @since 0.4.15
261      */
262     @NotNull
263     public JaxbBundle add(@NotNull final Element element) {
264         this.elements.add(element);
265         return this;
266     }
267 
268     /**
269      * Add new child with text value.
270      * @param nam The name of child
271      * @param txt The text
272      * @return The child bundle (use {@link #up()} on it in order to get back to
273      *  the parent bundle)
274      */
275     @NotNull
276     public JaxbBundle add(@NotNull final String nam,
277         @NotNull final String txt) {
278         final JaxbBundle child = new JaxbBundle(this, nam, txt);
279         this.children.add(child);
280         return child;
281     }
282 
283     /**
284      * Add XML attribute to this bundle.
285      * @param nam The name of attribute
286      * @param val The plain text value
287      * @return New bundle with set attribute
288      */
289     @NotNull
290     public JaxbBundle attr(@NotNull final String nam,
291         @NotNull final String val) {
292         this.attrs.put(nam, val);
293         return this;
294     }
295 
296     /**
297      * Add new link into {@code &lt;links&gt;} section.
298      * @param link The link to attach
299      * @return New bundle with a newly added link
300      * @since 0.4.10
301      */
302     @NotNull
303     public JaxbBundle link(@NotNull final Link link) {
304         this.links.add(link);
305         return this;
306     }
307 
308     /**
309      * Add new bundle.
310      * @param bundle The bundle to add
311      * @return New bundle with a newly added element
312      * @since 0.4.10
313      */
314     @NotNull
315     public JaxbBundle add(@NotNull final JaxbBundle bundle) {
316         this.children.add(bundle);
317         return this;
318     }
319 
320     /**
321      * Add new group.
322      * @param group The group
323      * @return New bundle with a newly added group of elements
324      * @since 0.4.10
325      */
326     @NotNull
327     public JaxbBundle add(@NotNull final JaxbBundle.Group<?> group) {
328         JaxbBundle holder = this;
329         for (final JaxbBundle bundle : group.bundles()) {
330             holder = holder.add(bundle);
331         }
332         return holder;
333     }
334 
335     /**
336      * Return parent bundle.
337      * @return The parent bundle
338      * @checkstyle MethodName (4 lines)
339      */
340     @NotNull
341     @SuppressWarnings("PMD.ShortMethodName")
342     public JaxbBundle up() {
343         if (this.parent == null) {
344             throw new IllegalArgumentException(
345                 "Already at the top, can't go #up()"
346             );
347         }
348         return this.parent;
349     }
350 
351     /**
352      * Convert this bundle into DOM/XML {@link Element}.
353      * @return The element
354      */
355     @NotNull
356     public Element element() {
357         if (this.parent != null) {
358             throw new IllegalArgumentException(
359                 "You can convert only top level JaxbBundle to DOM"
360             );
361         }
362         try {
363             return this.element(
364                 DocumentBuilderFactory.newInstance()
365                     .newDocumentBuilder()
366                     .newDocument()
367             );
368         } catch (final ParserConfigurationException ex) {
369             throw new IllegalStateException(ex);
370         }
371     }
372 
373     /**
374      * Create and return a DOM element, inside the document provided.
375      * @param doc The document
376      * @return The element just created
377      */
378     private Element element(final Document doc) {
379         final Element element = doc.createElement(this.name);
380         for (final Map.Entry<String, String> attr : this.attrs.entrySet()) {
381             element.setAttribute(attr.getKey(), attr.getValue());
382         }
383         for (final JaxbBundle child : this.children) {
384             element.appendChild(child.element(doc));
385         }
386         for (final Element child : this.elements) {
387             element.appendChild(doc.importNode(child, true));
388         }
389         if (!this.links.isEmpty()) {
390             final Element lnks = doc.createElement("links");
391             for (final Link link : this.links) {
392                 final Element lnk = doc.createElement("link");
393                 lnk.setAttribute("rel", link.getRel());
394                 lnk.setAttribute("href", link.getHref().toString());
395                 lnk.setAttribute("type", link.getType());
396                 lnks.appendChild(lnk);
397             }
398             element.appendChild(lnks);
399         }
400         element.appendChild(doc.createTextNode(this.content));
401         return element;
402     }
403 
404     /**
405      * Group.
406      * @param <T> Type of encapsulated elements
407      * @since 0.4.10
408      * @checkstyle AbstractClassName (2 lines)
409      */
410     @SuppressWarnings("PMD.AbstractNaming")
411     public abstract static class Group<T> {
412         /**
413          * Collection of objects.
414          */
415         private final transient Iterable<T> objects;
416         /**
417          * Public ctor.
418          * @param objs All objects
419          */
420         public Group(final Iterable<T> objs) {
421             this.objects = objs;
422         }
423         /**
424          * Convert an object to JaxbBundle.
425          * @param object The object to convert
426          * @return Bundle produced
427          */
428         public abstract JaxbBundle bundle(T object);
429         /**
430          * Fetch all bundles.
431          * @return All bundles
432          */
433         private Collection<JaxbBundle> bundles() {
434             final Collection<JaxbBundle> bundles = new LinkedList<JaxbBundle>();
435             for (final T object : this.objects) {
436                 bundles.add(this.bundle(object));
437             }
438             return bundles;
439         }
440     }
