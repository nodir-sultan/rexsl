57  /**
58   * JAXB group of elements.
59   *
60   * <p>A convenient utility class that enables on-fly creation of JAXB annotated
61   * collections of elements, for example:
62   *
63   * <pre> &#64;XmlRootElement
64   * public class Page {
65   *   &#64;XmlElement
66   *   public Object getEmployees() {
67   *     Collection&lt;Employee&gt; employees = this.retrieve();
68   *     return JaxbGroup.build(employees, "employee");
69   *   }
70   * }</pre>
71   *
72   * <p>It's even more convenient in combination with {@link JaxbBundle}, for
73   * example:
74   *
75   * <pre> &#64;XmlRootElement
76   * public class Page {
77   *   &#64;XmlElement
78   *   public Object getEmployee() {
79   *     return new JaxbBundle("employee")
80   *       .attr("age", "45")
81   *       .add("depts", JaxbGroup.build(this.depts(), "dept"))
82   *       .up()
83   *       .element();
84   *   }
85   * }</pre>
86   *
87   * <p>The class is immutable and thread-safe.
88   *
89   * @author Yegor Bugayenko (yegor@tpc2.com)
90   * @version $Id$
91   * @since 0.3.7
92   */
93  @XmlRootElement
94  @XmlAccessorType(XmlAccessType.NONE)
95  @ToString
96  public final class JaxbGroup {

99       * Classes already created before.
100      */
101     private static final ConcurrentMap<String, Class<?>> READY =
102         new ConcurrentHashMap<String, Class<?>>(0);

105      * Collection of elements.
106      */
107     @NotNull

113     public JaxbGroup() {

121      * Protected ctor, for on-fly instantiating (you're not supposed to use it).
122      * @param grp Group of elements

124     JaxbGroup(final Collection<?> grp) {
125         this.group = grp;

128     /**
129      * Creates a new JAXB-annotated collection of elements.
130      * @param grp Group of elements (JAXB-annotated)
131      * @param name Name of parent XML element
132      * @return JAXB-annotated object, just created
133      */
134     @NotNull
135     public static Object build(
136         @NotNull(message = "group can't be NULL") final Collection<?> grp,
137         @NotNull(message = "name can't be NULL") final String name) {
138         synchronized (JaxbGroup.class) {
139             final String mnemo = JaxbGroup.mnemo(grp.isEmpty(), name);
140             if (!JaxbGroup.READY.containsKey(mnemo)) {
141                 JaxbGroup.READY.put(
142                     mnemo,
143                     JaxbGroup.construct(JaxbGroup.types(grp), name)
144                 );
145             }
146             try {
147                 return JaxbGroup.READY.get(mnemo)
148                     .getDeclaredConstructor(Collection.class)
149                     .newInstance(grp);
150             } catch (NoSuchMethodException ex) {
151                 throw new IllegalStateException(ex);
152             } catch (InstantiationException ex) {
153                 throw new IllegalStateException(ex);
154             } catch (IllegalAccessException ex) {
155                 throw new IllegalStateException(ex);
156             } catch (InvocationTargetException ex) {
157                 throw new IllegalStateException(ex);
158             }
159         }
160     }
161 
162     /**
163      * Get group of elements.
164      * @return The collection

166     @XmlAnyElement(lax = true)

173     /**
174      * Create NEW class name.
175      * @param empty Is it an empty group?
176      * @param name Name of root element
177      * @return The name
178      */
179     private static String mnemo(final boolean empty, final String name) {
180         return String.format(
181             "%s$%s$%s",
182             JaxbGroup.class.getName(),
183             name,
184             // @checkstyle AvoidInlineConditionals (1 line)
185             empty ? "empty" : "full"
186         );
187     }

189     /**
190      * Construct new class.
191      * @param types Types used in the collection
192      * @param name Name of root element
193      * @return Class just created
194      */
195     private static Class<?> construct(final Collection<Class<?>> types,
196         final String name) {
197         try {
198             final CtClass ctc = PageBuilder.POOL.getAndRename(
199                 JaxbGroup.class.getName(),
200                 JaxbGroup.mnemo(types.isEmpty(), name)
201             );
202             final ClassFile file = ctc.getClassFile();
203             final AnnotationsAttribute attribute =
204                 AnnotationsAttribute.class.cast(
205                     file.getAttribute(
206                         AnnotationsAttribute.visibleTag
207                     )
208                 );
209             attribute.addAnnotation(JaxbGroup.xmlRootElement(file, name));
210             if (!types.isEmpty()) {
211                 attribute.addAnnotation(JaxbGroup.xmlSeeAlso(file, types));
212             }
213             final Class<?> cls = ctc.toClass();
214             ctc.defrost();
215             Logger.debug(
216                 JaxbGroup.class,
217                 "#construct('%s'): class %s constructed",
218                 name,
219                 cls.getName()
220             );
221             return cls;
222         } catch (NotFoundException ex) {
223             throw new IllegalStateException(ex);
224         } catch (CannotCompileException ex) {
225             throw new IllegalStateException(ex);
226         }
227     }

229     /**
230      * Find all types used in the collection.
231      * @param grp The collection
232      * @return List of types used there
233      */
234     private static Collection<Class<?>> types(final Collection<?> grp) {
235         final Collection<Class<?>> types = new HashSet<Class<?>>(grp.size());
236         for (final Object element : grp) {
237             types.add(element.getClass());
238         }
239         return types;
240     }

242     /**
243      * Create new <tt>XmlRootElement</tt> annotation.
244      * @param file Javassist file to work with
245      * @param name Name of root element
246      * @return The annotation
247      */
248     private static Annotation xmlRootElement(final ClassFile file,
249         final String name) {
250         final AnnotationsAttribute attribute = AnnotationsAttribute.class.cast(
251             file.getAttribute(AnnotationsAttribute.visibleTag)
252         );
253         final Annotation annotation = attribute.getAnnotation(
254             XmlRootElement.class.getName()
255         );
256         annotation.addMemberValue(
257             "name",
258             new StringMemberValue(name, file.getConstPool())
259         );
260         Logger.debug(
261             JaxbGroup.class,
262             "#xmlRootElement(.., '%s'): annotation created",
263             name
264         );
265         return annotation;
266     }

268     /**
269      * Create new <tt>XmlSeeAlso</tt> annotation.
270      * @param file Javassist file to work with
271      * @param types The class to refer to
272      * @return The annotation
273      */
274     @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
275     private static Annotation xmlSeeAlso(final ClassFile file,
276         final Collection<Class<?>> types) {
277         final Annotation annotation = new Annotation(
278             XmlSeeAlso.class.getName(),
279             file.getConstPool()
280         );
281         final ArrayMemberValue member = new ArrayMemberValue(
282             file.getConstPool()
283         );
284         final ClassMemberValue[] values = new ClassMemberValue[types.size()];
285         int pos = 0;
286         for (final Class<?> type : types) {
287             values[pos] = new ClassMemberValue(
288                 type.getName(),
289                 file.getConstPool()
290             );
291             pos += 1;
292         }
293         member.setValue(values);
294         annotation.addMemberValue("value", member);
295         Logger.debug(
296             JaxbGroup.class,
297             "#xmlSeeAlso(.., %d classes): annotation created",
298             types.size()
299         );
300         return annotation;
301     }
