47  /**
48   * JAXB-empowered object to XML converting utility.
49   *
50   * <p>The object has to be annotated with JAXB annotations
51   * in order to be convertable. Let's consider an example JAXB-annotated class:
52   *
53   * <pre> import javax.xml.bind.annotation.XmlAccessType;
54   * import javax.xml.bind.annotation.XmlAccessorType;
55   * import javax.xml.bind.annotation.XmlElement;
56   * import javax.xml.bind.annotation.XmlRootElement;
57   * &#64;XmlRootElement(name = "employee")
58   * &#64;XmlAccessorType(XmlAccessType.NONE)
59   * private static final class Employee {
60   *   &#64;XmlElement(name = "name")
61   *   public String getName() {
62   *     return "John Doe";
63   *   }
64   * }</pre>
65   *
66   * <p>Now you want to test how it works with real data after convertion
67   * to XML (in a unit test):
68   *
69   * <pre> import com.rexsl.test.JaxbConverter;
70   * import com.rexsl.test.XhtmlMatchers;
71   * import org.junit.Assert;
72   * import org.junit.Test;
73   * public final class EmployeeTest {
74   *   &#64;Test
75   *   public void testObjectToXmlConversion() throws Exception {
76   *     final Object object = new Employee();
77   *     Assert.assertThat(
78   *       JaxbConverter.the(object),
79   *       XhtmlMatchers.hasXPath("/employee/name[.='John Doe']")
80   *     );
81   *   }
82   * }</pre>
83   *
84   * @author Yegor Bugayenko (yegor@tpc2.com)
85   * @version $Id$
86   */
87  @ToString
88  @EqualsAndHashCode
89  @Loggable(Loggable.DEBUG)
90  public final class JaxbConverter {
91  
92      /**
93       * Private ctor, to avoid direct instantiation of the class.
94       */
95      private JaxbConverter() {
96          // intentionally empty
97      }
98  
99      /**
100      * Convert an object to XML.
101      *
102      * <p>The method will throw {@link AssertionError} if marshalling of
103      * provided object fails for some reason.
104      *
105      * <p>The name of the method is motivated by
106      * <a href="http://code.google.com/p/xml-matchers/">xmlatchers</a> project
107      * and their {@code XmlMatchers.the(String)} method. Looks like this name
108      * is short enough and convenient for unit tests.
109      *
110      * @param object The object to convert
111      * @param deps Dependencies that we should take into account
112      * @return DOM source/document
113      * @throws JAXBException If an exception occurs inside
114      * @todo #316 This "unchecked" warning should be removed somehow
115      */
116     @SuppressWarnings("unchecked")
117     public static Source the(@NotNull final Object object,
118         @NotNull final Class... deps)
119         throws JAXBException {
120         final Class<?>[] classes = new Class<?>[deps.length + 1];
121         classes[0] = object.getClass();
122         System.arraycopy(deps, 0, classes, 1, deps.length);
123         final JAXBContext ctx;
124         try {
125             ctx = JAXBContext.newInstance(classes);
126         } catch (final JAXBException ex) {
127             throw new IllegalArgumentException(ex);
128         }
129         final JAXBIntrospector intro = ctx.createJAXBIntrospector();
130         Object subject = object;
131         if (intro.getElementName(object) == null) {
132             subject = new JAXBElement<Object>(
133                 JaxbConverter.qname(object),
134                 (Class<Object>) object.getClass(),
135                 object
136             );
137         }
138         final Marshaller mrsh = JaxbConverter.marshaller(ctx);
139         final StringWriter writer = new StringWriter();
140         try {
141             mrsh.marshal(subject, writer);
142         } catch (final JAXBException ex) {
143             throw new AssertionError(ex);
144         }
145         final String xml = writer.toString();
146         return new StringSource(xml);
147     }
148 
149     /**
150      * Create marshaller.
151      * @param ctx The context
152      * @return Marshaller
153      * @throws JAXBException If an exception occurs inside
154      */
155     private static Marshaller marshaller(final JAXBContext ctx)
156         throws JAXBException {
157         final Marshaller mrsh = ctx.createMarshaller();
158         mrsh.setProperty(Marshaller.JAXB_FORMATTED_OUTPUT, Boolean.TRUE);
159         return mrsh;
160     }
161 
162     /**
163      * Get type name, if XmlType annotation is present (exception otherwise).
164      * @param obj The object
165      * @return Qualified name
166      * @see XmlElement#namespace()
167      */
168     private static QName qname(final Object obj) {
169         final XmlType type = XmlType.class.cast(
170             obj.getClass().getAnnotation(XmlType.class)
171         );
172         if (type == null) {
173             throw new AssertionError(
174                 Logger.format(
175                     // @checkstyle LineLength (1 line)
176                     "@XmlType or @XmlRootElement annotation required at %[type]s",
177                     obj
178                 )
179             );
180         }
181         QName qname;
182         if ("##default".equals(type.namespace())) {
183             qname = new QName(type.name());
184         } else {
185             qname = new QName(type.namespace(), type.name());
186         }
187         return qname;
188     }
