45  public final class CookieBuilderTest {

47      /**
48       * CookieBuilder can accept correct values.
49       * @throws Exception If there is some problem inside
50       */
51      @Test
52      @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
53      public void acceptsValidValues() throws Exception {
54          final String[] texts = {
55              "",
56              "text",
57              "some-text-to-accept(!)",
58          };
59          for (final String text : texts) {
60              new CookieBuilder(new URI("http://localhost/bar"))
61                  .name("some-name-of-cookie")
62                  .value(text)
63                  .build();
64          }
65      }
66  
67      /**
68       * CookieBuilder can reject incorrect values.
69       * @throws Exception If there is some problem inside
70       */
71      @Test
72      @SuppressWarnings("PMD.AvoidInstantiatingObjectsInLoops")
73      public void rejectsInvalidValues() throws Exception {
74          final String[] texts = {
75              " ",
76              ";",
77              "\\ backslash is not allowed",
78          };
79          for (final String text : texts) {
80              try {
81                  new CookieBuilder(new URI("http://localhost/foo"))
82                      .name("some-name")
83                      .value(text)
84                      .build();
85                  Assert.fail("Exception expected here");
86              } catch (IllegalArgumentException ex) {
87                  assert ex != null;
88              }
89          }
90      }
91  
92      /**
93       * CookieBuilder can build a valid cookie.
94       * @throws Exception If there is some problem inside
95       */
96      @Test
97      public void buildsCorrectCookie() throws Exception {
98          final String name = "some-cookie-name-6";
99          final String value = "some-value-of-it-6";
100         MatcherAssert.assertThat(
101             new CookieBuilder(new URI("http://google.com/6"))
102                 .name(name)
103                 .value(value)
104                 .build(),
105             Matchers.allOf(
106                 Matchers.hasToString(Matchers.containsString(name)),
107                 Matchers.hasProperty("maxAge", Matchers.lessThanOrEqualTo(0)),
108                 Matchers.hasProperty("domain", Matchers.equalTo("google.com")),
109                 Matchers.hasProperty("path", Matchers.equalTo("/6")),
110                 Matchers.hasProperty("name", Matchers.equalTo(name)),
111                 // @checkstyle MultipleStringLiterals (1 line)
112                 Matchers.hasProperty("value", Matchers.equalTo(value))
113             )
114         );
115     }
116 
117     /**
118      * CookieBuilder can build a cookie cleaning request.
119      * @throws Exception If there is some problem inside
120      */
121     @Test
122     public void buildsCookieDeletionRequest() throws Exception {
123         final String cookie = new CookieBuilder(new URI("http://localhost/f"))
124             .name("some-other-cookie-name")
125             .build()
126             .toString();
127         MatcherAssert.assertThat(
128             HttpCookie.parse(cookie).get(0),
129             Matchers.allOf(
130                 Matchers.hasProperty("value", Matchers.equalTo("")),
131                 new CustomMatcher<HttpCookie>("expired cookie") {
132                     @Override
133                     public boolean matches(final Object obj) {
134                         final HttpCookie cookie = HttpCookie.class.cast(obj);
135                         return cookie.hasExpired();
136                     }
137                 }
138             )
139         );
140     }
